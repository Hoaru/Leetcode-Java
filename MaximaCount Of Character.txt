import java.util.*;
public class Main {

    public static int solve(String s) {
        HashMap<Character, Integer> counts = new HashMap<>();
        HashMap<Character, Integer> maximaCount = new HashMap<>();
        HashSet<Character> lettersMax = new HashSet<>();
        int times = 0;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (i == 0) {
                counts.put(c, 1);
                maximaCount.put(c, 1);
                lettersMax.add(c);
                times += 1;
                maximaCount.put(c, 1);
            }
            else {
                counts.put(c, counts.getOrDefault(c, 0) + 1);
                if (counts.get(c) == times) {
                    lettersMax.add(c);
                    for (char e: lettersMax) {
                        maximaCount.put(e, maximaCount.getOrDefault(e, 0) + 1);
                    }
                }
                else if (counts.get(c) > times) {
                    times = counts.get(c);
                    lettersMax.clear();
                    lettersMax.add(c);
                    maximaCount.put(c, maximaCount.get(c) + 1);
                }
                else {
                    for (char e: lettersMax) {
                        maximaCount.put(e, maximaCount.get(e) + 1);
                    }
                }
            }
        }
        int res = 0;
        for (int i : maximaCount.values()) {
            res = Math.max(res, i);
        }
        return res;
    }
    public static void main(String[] args) {
        System.out.print(solve("adbcbcbcc"));
    }
}
