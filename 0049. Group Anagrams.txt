class Solution {
    public List<List<String>> groupAnagrams(String[] strs) {
        List<List<String>> res = new ArrayList<>();
        HashMap<String, List> map = new HashMap<>();
        for(String s : strs){
            // List<Character> clist = s.toCharArray();
            char[] clist = s.toCharArray();
            Arrays.sort(clist);
            String alphaOrder = new String(clist);
            if(!map.containsKey(alphaOrder)){
                map.put(alphaOrder, new ArrayList<String>());
            }
            map.get(alphaOrder).add(s);
        }
        return new ArrayList(map.values());
    }
}
