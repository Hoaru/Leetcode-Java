class MinStack {
    Stack<Integer> st1;
    Stack<Integer> st2;

    public MinStack() {
      st1 = new Stack<>();
      st2 = new Stack<>();
    }
    
    public void push(int val) {
        st1.add(val);
        if(st2.isEmpty() || st2.peek() >= val) {
            st2.add(val);
        }
    }
    
    public void pop() {
        int x = st1.pop();
        if(!st2.isEmpty() && st2.peek() == x) {
            st2.pop();
        }
    }
    
    public int top() {
        return st1.peek();
    }
    
    public int getMin() {
        return st2.peek();
    }
}
