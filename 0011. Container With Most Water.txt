class Solution {
    public int maxArea(int[] height) {
        int max = 0;
        for (int l = 0, r = height.length - 1; l < r;) {
            int curr;
            if (height[l] - height[r] >= 0) {
                curr = (r - l) * height[r--];
            }
            else {
                curr = (r - l) * height[l++];
            }
            max = Math.max(curr, max);
        }
        return max;
    }
}
